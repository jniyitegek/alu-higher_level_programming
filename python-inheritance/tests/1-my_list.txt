# Test 1: Basic sorting
>>> my_list = MyList()
>>> my_list.append(3)
>>> my_list.append(1)
>>> my_list.append(2)
>>> my_list.print_sorted()
[1, 2, 3]

# Test 2: Empty list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

# Test 3: Single element
>>> my_list = MyList()
>>> my_list.append(42)
>>> my_list.print_sorted()
[42]

# Test 4: Already sorted list
>>> my_list = MyList()
>>> my_list.append(10)
>>> my_list.append(20)
>>> my_list.append(30)
>>> my_list.print_sorted()
[10, 20, 30]

# Test 5: Negative numbers
>>> my_list = MyList()
>>> my_list.append(-5)
>>> my_list.append(-1)
>>> my_list.append(-3)
>>> my_list.print_sorted()
[-5, -3, -1]

# Test 6: Duplicate elements
>>> my_list = MyList()
>>> my_list.append(5)
>>> my_list.append(5)
>>> my_list.append(3)
>>> my_list.print_sorted()
[3, 5, 5]
