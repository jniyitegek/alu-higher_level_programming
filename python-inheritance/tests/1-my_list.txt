# Test 1: Basic sorting
my_list = MyList()
my_list.append(3)
my_list.append(1)
my_list.append(2)
my_list.print_sorted()  # Expected output: [1, 2, 3]

# Test 2: Empty list
my_list = MyList()
my_list.print_sorted()  # Expected output: []

# Test 3: Single element
my_list = MyList()
my_list.append(42)
my_list.print_sorted()  # Expected output: [42]

# Test 4: Already sorted list
my_list = MyList()
my_list.append(10)
my_list.append(20)
my_list.append(30)
my_list.print_sorted()  # Expected output: [10, 20, 30]

# Test 5: Negative numbers
my_list = MyList()
my_list.append(-5)
my_list.append(-1)
my_list.append(-3)
my_list.print_sorted()  # Expected output: [-5, -3, -1]
